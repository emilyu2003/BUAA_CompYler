int array[100];

int factorial(int n) {
    printf(":%d\n", n);
    if (array[n] != 0) {
        return array[n];
    } else {
        array[n - 1] = factorial(n - 1);
        array[n] = array[n - 1] * n;
        return array[n];
    }
    return 0;
}

int main() {
    int int_c = 8;

    int i = 0;
    while (i < 100) {
        if (i == 0 || i == 1) {
            array[i] = 1;
        } else {
            array[i] = 0;
        }
        i = i + 1;
    }
    factorial(5);
    return 0;
}
